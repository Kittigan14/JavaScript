function calculateEquation(equation) {
  let currentNumber = '';
  let currentOperator = '+';
  let result = 0;

  for (const char of equation) {
    if ('+-*/'.includes(char)) {
      result = evalOperation(result, Number(currentNumber), currentOperator);
      currentOperator = char;
      currentNumber = '';
    } else {
      currentNumber += char;
    }
  }

  result = evalOperation(result, Number(currentNumber), currentOperator);

  return result;
}

function evalOperation(prevResult, currentNumber, operator) {
  switch (operator) {
    case '+':
      return prevResult + currentNumber;
    case '-':
      return prevResult - currentNumber;
    case '*':
      return prevResult * currentNumber;
    case '/':
      return prevResult / currentNumber;
    default:
      return prevResult;
  }
}

const equationResult = calculateEquation("1+2*3-4/2");
console.log(equationResult); // ผลลัพธ์: 5
